/*
 * 求x的n次幂（快速幂方式） 时间复杂度：O(logn)
 */
public static int pow(int x, int n) {
    int res = 1;
    while (n > 0) {
        if ((n & 1) == 1) {
            res *= x;
        }
        n >>= 1;
        x = x * x;
    }
    return res;
}
